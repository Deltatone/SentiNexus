from bs4 import BeautifulSoup # Load the original index HTML file with open("index (1).html", "r", encoding="utf-8") as file: original_html = file.read() # Parse the original HTML to extract header and footer soup = BeautifulSoup(original_html, "html.parser") # Extract header and footer content header = soup.find("header") footer = soup.find("footer") # Create the new HTML content new_html = """ SentiLedger Login {header}

SentiLedger

Forgot Password?
{footer} """.format(header=str(header) if header else "", footer=str(footer) if footer else "") # Save the new HTML page with open("sentiledger_login.html", "w", encoding="utf-8") as file: file.write(new_html) print("The login page 'sentiledger_login.html' has been created successfully.")
